/*jslint esnext:true, browser:true*/
/*globals App, Question*/
class Courtes extends Question {
	/**
	 * Constructor
	 * @param {Qr}          qr  - L'objet Qr auquel est associée la question
	 * @param {HTMLElement} dom - L'élément HTML à traiter
	 */
	constructor(qr, dom) {
		super(qr, dom);
	}
	init() {
		this.aleaSelecteur = "div.qr-body>div";
		this.alea = true;
		this.largeur = null;
		this.largeurEnonce = null;
		this.inverse = false;
		this.numero = "alpha";
		super.init();
	}
	static init() {
		this.className = 'courtes';
		this.ajouterStyle ({
			all: {
				'*.qr-body': {
					'display': 'flex',
					'flex-direction': 'column',
					'justify-content': 'space-between',
					'&>div': {
					    'display': 'flex',
					    'margin-top': '2mm',
						'align-items': 'center',
						'&.off': {
							'display': 'none'
						},
						'&>*+*': {
							'margin-left': '1ch'
						},
						'&>*:nth-child(1)': {
							'flex': '0 1 auto'
						},
						'&>*:nth-child(2)': {
							'border-style': 'none none solid',
							'flex': '1 0 auto',
							'display': 'flex',
							'align-items': 'inherit'
						}
					},
					'*.qr-reponse': {
//						'align-self': 'flex-end',
					    'margin-left': '0',
					    'margin-right': '1ch',
					    'min-height': '6mm'
					}
				},
				'*.qr-body.qr-inverse': {
					'&>div': {
					    'margin-top': '2mm'
					},
					'*.qr-enonce': {
					    'order': '1'
					},
					'*.qr-reponse': {
					    'margin-left': '0',
					    'margin-right': '1ch',
					    'min-height': '6mm'
					}
				}
			}
		});
	}
	get inverse() {
		return this._inverse || false;
	}
	set inverse(val) {
		this._inverse = App.parseBool(val);
	}
	traiterDom() {
		var div, enonces, i, n, enonce;
		super.traiterDom();
		div = document.createElement("div");
		div.classList.add('qr-body');
		if (this.inverse) {
			this.body.classList.add('qr-inverse');
		}
		if (this.numero !== "none") {
			this.body.classList.add('numero');
			this.body.classList.add(this.numero);
		}
		enonces = this.body.children;
		for (i = 0, n = enonces.length; i < n; i += 1) {
			enonce = enonces[i];
			this.body.replaceChild(this.dom_rangee(enonce), enonce);
		}
		return this;
	}
	dom_rangee(question) {
		var resultat, reponse, enonce, solution;
		resultat = document.createElement('div');
		App.transferClasses(question, resultat);
		// La cellule de la question
		enonce = resultat.appendChild(document.createElement('span'));
		enonce.classList.add("qr-enonce");
		enonce.innerHTML = question.innerHTML;
		// La cellule de la reponse
		reponse = resultat.appendChild(document.createElement('span'));
		reponse.setAttribute('class', 'qr-reponse');
		solution = reponse.appendChild(document.createElement('span'));
		solution.setAttribute('class', 'qr-solution');
		solution.innerHTML = question.getAttribute('data-solution');
		if (this.largeurEnonce) {
			enonce.style.flex = "0 0 " + this.largeurEnonce;
		}
		if (this.largeur) {
			reponse.style.flex = "0 0 " + this.largeur;
		}
//		if (this.inverse) {
//			enonce.style.order = 1;
//		}
		return resultat;
	}
}
Courtes.init();
